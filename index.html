        async function translateWithAI(text) {

            const prompt = `Translate the following hotel room type to professional English. Provide only the translated text, nothing else. If it's already in English, just return the original text. Text: "${text}"`;

            const payload = { contents: [{ role: "user", parts: [{ text: prompt }] }] };

            const apiKey = ""; // Handled by environment

            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;

            try {

                const response = await fetch(apiUrl, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) });

                if (!response.ok) throw new Error(`API request failed: ${response.status}`);

                const result = await response.json();

                if (result.candidates?.[0]?.content?.parts?.[0]?.text) {

                    return { success: true, text: result.candidates[0].content.parts[0].text.trim(), method: 'AI' };

                }

                throw new Error("Invalid API response format");

            } catch (error) {

                console.warn("AI Translation failed:", error.message);

                return { success: false, text: text };

            }

        }


        function translateWithDict(text) {

            const cleanText = text.trim();

            if (roomTypeDictionary[cleanText]) {

                return { success: true, text: roomTypeDictionary[cleanText], method: 'Dictionary' };

            }

            let combined = '';

            if (cleanText.includes('海景')) combined += 'Sea View ';

            if (cleanText.includes('山景')) combined += 'Mountain View ';

            if (cleanText.includes('城景')) combined += 'City View ';

            if (cleanText.includes('大床')) combined += 'King Room';

            else if (cleanText.includes('雙床') || cleanText.includes('双床')) combined += 'Twin Room';

            else if (cleanText.includes('單人') || cleanText.includes('单人')) combined += 'Single Room';

            if (combined) return { success: true, text: combined.trim(), method: 'Dictionary' };

            

            return { success: false, text: text };

        }

        }

        function copyToClipboard(text, buttonElement) {

            textArea.style.position = 'absolute';

            textArea.style.left = '-9999px';

                const originalText = buttonElement.innerText;

                buttonElement.innerText = translations[currentLang].copied;

                buttonElement.classList.add('copied');

            } catch (err) {

            }

            document.body.removeChild(textArea);

        }

            dateErrorDiv.textContent = '';

            resultArea.innerHTML = '';


            const checkInDate = document.getElementById('check-in').value;

            const checkOutDate = document.getElementById('check-out').value;

            

            if (checkInDate && checkOutDate && new Date(checkOutDate) <= new Date(checkInDate)) {

                dateErrorDiv.textContent = translations[currentLang].dateError;

                return;

            }

            

            const roomEntries = roomsContainer.querySelectorAll('.room-entry');

            if (roomEntries.length === 0) {

                dateErrorDiv.textContent = translations[currentLang].roomError;

                return;

            }


            generateBtn.parentElement.disabled = true;

            generateBtn.parentElement.innerHTML = `<div class="loader"></div><span data-translate="generating">${translations[currentLang].generating}</span>`;


            const hotelName = document.getElementById('hotel-name').value;

            const hotelEmail = document.getElementById('hotel-email').value;

            const guestName = document.getElementById('guest-name').value;

            const bookingWebsite = document.getElementById('booking-website').value;

            const bookingRef = document.getElementById('booking-ref').value;

            

            let totalGuests = 0;

            const roomPromises = Array.from(roomEntries).map(async (entry) => {

                const roomTypeInput = entry.querySelector('input[name="room-type"]').value;

                const numGuests = parseInt(entry.querySelector('input[name="num-guests"]').value, 10);

                totalGuests += numGuests;

                const translationResult = await translateRoomType(roomTypeInput);

                return { ...translationResult, guests: numGuests };

            });


            const roomDetails = await Promise.all(roomPromises);

            const roomListString = roomDetails.map(room => `- ${room.text} (${room.guests} guest(s))`).join('\n');


            const subject = `Booking Confirmation Request - ${guestName} - Arrival: ${checkInDate}`;

            const body = `Dear ${hotelName} Reservation Team,\n\nI am writing to kindly request a confirmation for the following reservation, which was made through ${bookingWebsite}.\n\nGuest Name: ${guestName}\nBooking Reference Number: ${bookingRef}\nCheck-in Date: ${checkInDate}\nCheck-out Date: ${checkOutDate}\n\nThe booking details are as follows:\n${roomListString}\n\nTotal Number of Guests: ${totalGuests}\n\nI would be most grateful if you could verify the details above and confirm that the reservation is secured in your system.\n\nThank you for your time and assistance. I look forward to hearing from you soon.\n\nSincerely,\n${guestName}`;


            const encodedSubject = encodeURIComponent(subject);

            const encodedBody = encodeURIComponent(body);

            const mailtoLink = `mailto:${hotelEmail}?subject=${encodedSubject}&body=${encodedBody}`;


            resultArea.innerHTML = `

                <div class="bg-white p-6 rounded-lg shadow-md mt-6 border border-slate-200 fade-in">

                    <h3 class="text-xl font-semibold text-slate-800 mb-4">${translations[currentLang].emailReady}</h3>

                    <div class="space-y-6">

                        <div>

                            <h4 class="text-sm font-bold text-slate-500 mb-2 uppercase tracking-wider">${translations[currentLang].subject}</h4>

                            <div class="flex items-center gap-4">

                                <p id="email-subject" class="flex-grow bg-slate-100 p-3 rounded-md text-left text-slate-700">${subject}</p>

                                <button onclick="copyToClipboard(document.getElementById('email-subject').innerText, this)" class="copy-button bg-slate-200 hover:bg-slate-300 text-slate-700 font-semibold py-2 px-4 rounded-lg transition">${translations[currentLang].copy}</button>

                            </div>

                        </div>

                        <div>

                            <h4 class="text-sm font-bold text-slate-500 mb-2 uppercase tracking-wider">${translations[currentLang].body}</h4>

                            <div class="flex items-start gap-4">

                                <pre id="email-body" class="flex-grow bg-slate-100 p-3 rounded-md text-left whitespace-pre-wrap text-sm leading-relaxed text-slate-700">${body.replace(/\n/g, '<br>')}</pre>

                                <button onclick="copyBody(this)" class="copy-button bg-slate-200 hover:bg-slate-300 text-slate-700 font-semibold py-2 px-4 rounded-lg transition">${translations[currentLang].copy}</button>

                            </div>

                        </div>

                        <div class="border-t border-slate-200 pt-6 text-center">

                             <p class="text-slate-500 mb-4">${translations[currentLang].openInApp}</p>

                             <a href="${mailtoLink}" target="_blank" class="inline-block bg-indigo-600 hover:bg-indigo-700 text-white font-bold py-3 px-8 rounded-lg text-lg transition-transform transform hover:scale-105 shadow-md">

                                ${translations[currentLang].openBtn}

                            </a>

                        </div>

                    </div>

                </div>

            `;

            

            generateBtn.parentElement.disabled = false;

            generateBtn.parentElement.innerHTML = `<span data-translate="generateEmail">${originalBtnText}</span>`;

            resultArea.scrollIntoView({ behavior: 'smooth' });

        });
